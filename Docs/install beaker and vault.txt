# create_vault_service


cat > /etc/systemd/system/consul.service <<EOF
[Unit]
Description=Consul
Documentation=https://www.consul.io/

[Service]
ExecStart=/usr/bin/consul agent -server -ui -data-dir=/tmp/consul -bootstrap-expect=1 -node=vault -bind=192.168.1.28 -config-dir=/etc/consul.d/
ExecReload=/bin/kill -HUP $MAINPID
LimitNOFILE=65536

[Install]
WantedBy=multi-user.target
EOF

mkdir -p /etc/consul.d
cat > /etc/consul.d/ui.json <<EOF
{
  "addresses": {
    "http": "0.0.0.0"
  }
}
EOF

systemctl daemon-reload
systemctl start consul
systemctl enable consul






cp vault /usr/local/bin
mkdir -p /etc/vault
cat > /etc/vault/config.hcl <<EOF
storage "consul" {
  address = "127.0.0.1:8500"
  path    = "vault/"
}

listener "tcp" {
 address     = "127.0.0.1:8200"
 tls_disable = 1
}

ui = true
EOF

cat >  /etc/systemd/system/vault.service <<EOF
[Unit]
Description=Vault
Documentation=https://www.vault.io/

[Service]
ExecStart=/usr/bin/vault server -config=/etc/vault/config.hcl
ExecReload=/bin/kill -HUP $MAINPID
LimitNOFILE=65536

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl start vault
systemctl enable vault

export VAULT_ADDR=http://127.0.0.1:8200

vault operator init
